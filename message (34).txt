getgenv().AimPart = "HumanoidRootPart"
getgenv().AimRadius = 500
getgenv().PredictMovement = true
getgenv().PredictionVelocity = 6.7

local Players = game:GetService("Players")
local RunService = game:GetService("RunService")
local Camera = workspace.CurrentCamera
local Client = Players.LocalPlayer
local AimlockTarget = nil
local MousePressed = false

local function getNearestTarget()
    local nearestPlayer = nil
    local nearestDistance = getgenv().AimRadius

    for _, player in pairs(Players:GetPlayers()) do
        if player ~= Client and player.Character and player.Character:FindFirstChild(getgenv().AimPart) then
            local targetPart = player.Character[getgenv().AimPart]
            local distance = (targetPart.Position - Camera.CFrame.Position).magnitude
            if distance < nearestDistance then
                nearestDistance = distance
                nearestPlayer = player
            end
        end
    end

    return nearestPlayer
end

TextButton.MouseButton1Click:Connect(function()
    if not MousePressed then
        MousePressed = true
        AimlockTarget = getNearestTarget()
    else
        AimlockTarget = nil
        MousePressed = false
    end
end)

TextBox:GetPropertyChangedSignal("Text"):Connect(function()
    local predictionValue = tonumber(TextBox.Text)
    if predictionValue then
        getgenv().PredictionVelocity = predictionValue
    end
end)

RunService.RenderStepped:Connect(function()
    if AimlockTarget and AimlockTarget.Character then
        local targetPart = AimlockTarget.Character:FindFirstChild(getgenv().AimPart)
        if targetPart then
            local targetPosition = targetPart.Position
            if getgenv().PredictMovement then
                targetPosition = targetPosition + targetPart.Velocity / getgenv().PredictionVelocity
            end
            
            Camera.CFrame = CFrame.new(Camera.CFrame.Position, targetPosition)
        end
    end
end)
